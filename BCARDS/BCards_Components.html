① Análise — ctx-icon: variação ou instância independente?
A classe ctx-icon punch-winner visible phase-select icon-flipped icon-flipped-v que você está vendo não é uma instância independente — é a mesma div base acumulando modificadores. O componente tem uma única estrutura no DOM para cada jogador (#ctx-icon-p1 e #ctx-icon-p2), e as classes são trocadas pelo JS via updateCtxIcon(), showPunchIcon() etc.

Conclusão: existe uma única div .ctx-icon por jogador. Todas as variações (select, confirm, battle, punch-winner, punch-loser, push) são modificadores CSS empilhados sobre o mesmo elemento. O SVG interno é clonado dinamicamente de templates <svg id="svg-icon-show|ok|punch" style="display:none">. A div nunca carrega mais de um SVG ao mesmo tempo — applyIcon() faz el.innerHTML = '' antes de injetar o clone.
Consequência para o componente modular: a lógica de "qual SVG + quais classes" está espalhada em 4 funções (updateCtxIcon, showOkIcons, hideOkIcon, showPunchIcon). Um componente modular centraliza isso em um único objeto com método .set(spec).
Sobre o widget de animação na jornada: é viável. Cada keyframe do ctx-icon já é uma animação CSS nomeada (iconPulse, punchWinner, punchEnter, punchImpactP1/P2, punchLoserPushedUp/Down). Um widget poderia expor sliders de duration, easing e translateY offset via CSS custom properties injetadas no :root. Complexidade adicionada: média. Não quebra nada do engine atual pois as animações são CSS puras — o JS não controla os valores internos dos keyframes. Fica em aberto para a próxima iteração.
② CtxIconComponent — ícone contextual modular
O componente substitui as 4 funções globais (updateCtxIcon, showOkIcons, hideOkIcon, showPunchIcon) por uma fábrica com API declarativa. Cada instância controla um único .ctx-icon e expõe .set(spec), .clear() e .animate(name). A integração no HTML é zero-invasiva: substitui o bloco ctx-icon-system.

/* ════════════════════════════════════════════════════════════════
   CTX ICON COMPONENT — Elemental Clash v29
   ════════════════════════════════════════════════════════════════

   FACTORY: CtxIcon(elementId, options?)
   Cria uma instância que controla um único .ctx-icon no DOM.

   options = {
     svgTemplates : { show:'svg-icon-show', ok:'svg-icon-ok', punch:'svg-icon-punch' },
     size         : null,      // string CSS ou null (usa clamp do CSS)
     position     : 'right',   // 'right' | 'left' | { top, right, bottom, left }
     zIndex       : 25,
   }

   SPEC para .set(spec):
   {
     svg     : 'show' | 'ok' | 'punch' | string,  // chave em svgTemplates
     phase   : 'select' | 'confirm' | 'battle',   // classe CSS phase-*
     flip    : false,    // scaleX(-1) no SVG
     flipV   : false,    // scaleY(-1) no SVG
     anim    : 'pulse' | 'punch-winner' | 'punch-entering' |
               'striking-p1' | 'striking-p2' |
               'pushed-up' | 'pushed-down' | null,
   }
   ════════════════════════════════════════════════════════════════ */
'use strict';

function CtxIcon(elementId, opts = {}) {
  const el = document.getElementById(elementId);
  if (!el) { console.warn(`[CtxIcon] Elemento não encontrado: "${elementId}"`); return null; }

  const defaults = {
    svgTemplates: {
      show  : 'svg-icon-show',
      ok    : 'svg-icon-ok',
      punch : 'svg-icon-punch',
    },
    size     : null,
    position : 'right',
    zIndex   : 25,
  };
  const cfg = Object.assign({}, defaults, opts);

  // Aplica overrides de tamanho/posição via style inline
  _applyLayout();

  // ── Classes de animação mapeadas ──────────────────────────────
  const ANIM_CLASSES = {
    'pulse'        : [],  // animação padrão de iconPulse (via .visible)
    'punch-winner' : ['punch-winner'],
    'punch-entering': ['punch-winner', 'punch-entering'],
    'striking-p1'  : ['punch-winner', 'punch-striking-p1'],
    'striking-p2'  : ['punch-winner', 'punch-striking-p2'],
    'pushed-up'    : ['punch-loser', 'punch-pushed-up'],
    'pushed-down'  : ['punch-loser', 'punch-pushed-down'],
  };

  // Todas as classes removíveis (não inclui a base .ctx-icon)
  const ALL_MODIFIER_CLASSES = [
    'visible', 'phase-select', 'phase-confirm', 'phase-battle',
    'icon-flipped', 'icon-flipped-v',
    'punch-winner', 'punch-loser',
    'punch-entering', 'punch-striking-p1', 'punch-striking-p2',
    'punch-pushed-up', 'punch-pushed-down',
  ];

  // ── LAYOUT ───────────────────────────────────────────────────
  function _applyLayout() {
    if (cfg.size) {
      el.style.width  = cfg.size;
      el.style.height = cfg.size;
    }

    el.style.zIndex = cfg.zIndex;

    if (typeof cfg.position === 'string') {
      // 'right' (padrão) ou 'left'
      el.style.right = cfg.position === 'left' ? 'auto' : 'clamp(10px, 2.5vw, 26px)';
      el.style.left  = cfg.position === 'left' ? 'clamp(10px, 2.5vw, 26px)' : 'auto';
      el.style.top   = '50%';
    } else {
      // objeto { top, right, bottom, left } — posicionamento livre
      Object.assign(el.style, cfg.position);
    }
  }

  // ── CLEAR — reseta o elemento ao estado invisível ─────────────
  function clear() {
    el.classList.remove(...ALL_MODIFIER_CLASSES);
    el.innerHTML = '';
    el.style.animation = '';
  }

  // ── INJECT SVG — clona template e injeta no container ─────────
  function _injectSvg(svgKey) {
    const tmplId = cfg.svgTemplates[svgKey] || svgKey; // aceita id direto
    const tmpl = document.getElementById(tmplId);
    if (!tmpl) { console.warn(`[CtxIcon] SVG template não encontrado: "${tmplId}"`); return; }
    const clone = tmpl.cloneNode(true);
    clone.removeAttribute('id');
    clone.removeAttribute('style');
    el.innerHTML = '';
    el.appendChild(clone);
  }

  // ── SET — aplica um spec completo ─────────────────────────────
  function set(spec) {
    if (!spec) { clear(); return; }

    // 1) Limpa tudo
    clear();

    // 2) Injeta SVG
    if (spec.svg) _injectSvg(spec.svg);

    // 3) Classe de fase
    if (spec.phase) el.classList.add(`phase-${spec.phase}`);

    // 4) Flips no SVG
    if (spec.flip)  el.classList.add('icon-flipped');
    if (spec.flipV) el.classList.add('icon-flipped-v');

    // 5) Animação
    const animKey = spec.anim || 'pulse';
    const animCls = ANIM_CLASSES[animKey] || [];
    animCls.forEach(c => el.classList.add(c));

    // 6) Torna visível por último (dispara a transição de opacity)
    el.classList.add('visible');
  }

  // ── ANIMATE — troca apenas a animação sem recriar o SVG ───────
  // Útil para transições fight-impact: punch-entering → striking → loop
  function animate(animKey) {
    const animCls = ANIM_CLASSES[animKey];
    if (!animCls) return;

    // Remove classes de animação atuais
    const animAllCls = Object.values(ANIM_CLASSES).flat();
    el.classList.remove(...animAllCls);

    // Force reflow para reiniciar a animação CSS
    void el.offsetWidth;

    animCls.forEach(c => el.classList.add(c));
  }

  // ── UPDATE LAYOUT — permite reposicionar após criação ─────────
  function updateLayout(newOpts) {
    Object.assign(cfg, newOpts);
    _applyLayout();
  }

  return { set, clear, animate, updateLayout, el };
}
